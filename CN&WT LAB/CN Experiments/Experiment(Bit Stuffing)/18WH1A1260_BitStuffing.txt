#include <stdio.h>

int main()
{
    int n,count=0;
    printf("Enter number of frames:");
    scanf("%d",&n);
    int bits[100],i,j=0,stuff[100];
    printf("Enter bits:\n");
    for(i=0;i<n;i++){
        scanf("%d",&bits[i]);
    }
    for(i=0;i<n;i++){
        if(bits[i] == 1){
            count++;
        }else{
            count=0;
        }
       stuff[j] = bits[i];
       j++;
       if(count == 5 && i != n-1 ){
           count=0;
           stuff[j]=0;
           j++;
       }
    }
    printf("At sender\n");
    printf("After Bit stuffing : 01111110 ");
    for(i=0;i<j;i++){
        printf("%d",stuff[i]);
    }
    printf(" 01111110\n");
    printf("At receiver\n");
    printf("Data received is:01111110 ");
    for(i=0;i<j;i++){
        printf("%d",stuff[i]);
    }
    printf(" 01111110\n");
    printf("After de-stuffing:\n");
    count=0;
    for(i=0;i<j;i++){
        if(stuff[i] == 1){
            count++;
        }else{
            count=0;
        }
        printf("%d",stuff[i]);
        if(count == 5){
            i++;
        }
    }
    printf("\n");
    
    return 0;
}
